write a query to display the below columns
 C_CUSTKEY,C_NAME,O_ORDERKEY,O_ORDERSTATUS,O_totalprice,l_linenumber,l_quantity; (CUSTOMER, ORDERS, LINEITEM)
A. SELECT C.C_CUSTKEY, C.C_NAME, O.O_ORDERKEY, O.O_ORDERSTATUS, O.O_TOTALPRICE, L.L_LINENUMBER, L.L_QUANTITY FROM 
SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.CUSTOMER C INNER JOIN SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.ORDERS O ON(C.C_CUSTKEY = O.O_CUSTKEY) INNER JOIN SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.LINEITEM L ON(O.O_ORDERKEY = L.L_ORDERKEY) LIMIT 10;
 
 write a query to display the below columns with filter condtions O_ORDERDATE,O_TOTALPRICE,O_ORDERPRIORITY,C_NAME,N_NAME O_TOTALPRICE less than 10000 O_ORDERPRIORITY should be 1-URGENT OR 2-HIGH
A. SELECT O.O_ORDERDATE, O.O_TOTALPRICE, O.O_ORDERPRIORITY, C.C_NAME, N.N_NAME FROM SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.ORDERS O INNER JOIN SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.CUSTOMER C ON(O.O_CUSTKEY = C.C_CUSTKEY) INNER JOIN SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.NATION N ON(C.C_NATIONKEY = N.N_NATIONKEY) WHERE O.O_TOTALPRICE < 100000 AND O.O_ORDERPRIORITY IN ('1-URGENT', '2-HIGH') LIMIT 10;

 
 
  write a query to display the below columns with filter condtions
O_ORDERKEY,C_NAME,N_NAME,O_ORDERDATE,O_TOTALPRICE,l_quantity,L_LINESTATUS,L_SHIPDATE,P_PARTKEY,P_BRAND,P_CONTAINER,R_NAME l_quantity range should be 10-50 L_SHIPDATE range should be 1994-11-04  to 1998-11-04 
l_shipmode should be MAIL or RAIL O_TOTALPRICE greater than 50000 l_discount greater than 0.08
(OREDERS, CUSTOMER, NATION, LINEITEM, PART, REGION
A. SELECT O.O_ORDERKEY, C.C_NAME, N.N_NAME, O.O_ORDERDATE, O.O_TOTALPRICE, L.L_DISCOUNT, L.L_QUANTITY, L.L_LINESTATUS, L.L_SHIPDATE, L.L_SHIPMODE, P.P_PARTKEY, P.P_BRAND, P.P_CONTAINER, R.R_NAME FROM SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.REGION R INNER JOIN SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.NATION N ON(R.R_REGIONKEY = N.N_NATIONKEY) INNER JOIN SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.CUSTOMER C ON(N.N_NATIONKEY = C.C_NATIONKEY) INNER JOIN SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.ORDERS O ON(C.C_CUSTKEY = O.O_CUSTKEY) INNER JOIN SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.LINEITEM L ON(O.O_ORDERKEY = L.L_ORDERKEY) INNER JOIN SNOWFLAKE_SAMPLE_DATA.TPCH_SF1.PART P ON(L.L_PARTKEY = P.P_PARTKEY) WHERE L.L_QUANTITY BETWEEN 10 AND 50 
AND L.L_SHIPDATE BETWEEN '1994-11-04' AND '1998-11-04' 
AND L.L_SHIPMODE IN ('MAIL', 'RAIL') 
AND O.O_TOTALPRICE > 50000 
AND L.L_DISCOUNT > 0.08 LIMIT 10;
